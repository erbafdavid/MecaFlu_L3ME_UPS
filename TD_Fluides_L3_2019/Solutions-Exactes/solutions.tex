\begin{minipage}[c]{4.5cm}
\frimage{poiseuille-plan.pstex_t}
\end{minipage}
&
\begin{minipage}[c]{2cm}
\[
\dpdt{\vec{u}} = \vec{0}
\]
\[
\dpdx{\vec{u}} = \dpdz{\vec{u}} = \vec{0}
\]
\vskip 0.5cm
\end{minipage}
&
\begin{minipage}[c]{5.5cm}
\[
\frac{dp}{dx} = C^{te}
\quad
\mbox{et} \quad
\vec{u} = u(y) \, \vec{e}_x 
\]
\[
\mbox{avec} \quad
u(y) = \frac{1}{2\mu} \frac{dp}{dx} y (h-y)
\]
\end{minipage}
&
\begin{minipage}[c]{4cm}
Poiseuille plan
\end{minipage}
\\
\hline \hline
\begin{minipage}[c]{4.5cm}
\frimage{couette-poiseuille.pstex_t}
\end{minipage}
&
\begin{minipage}[c]{2cm}
\[
\dpdt{\vec{u}} = \vec{0}
\]
\[
\dpdx{\vec{u}} = \dpdz{\vec{u}} = \vec{0}
\]
\vskip 0.5cm
\end{minipage}
&
\begin{minipage}[c]{5.5cm}
\[
\frac{dp}{dx} = C^{te}
\quad
\mbox{et} \quad
\vec{u} = u(y) \, \vec{e}_x 
\]
\[ \mbox{avec} \quad
u(y) = U \left[ 
\frac{y}{h} -  K \frac{y}{h} \left ( 1 - \frac{y}{h} \right ) \right ]
\] 
\[
\mbox{o\`u}  \quad K =\frac{h^2}{2\mu U} \ddx{p}
\]
\vskip 0.5cm
\end{minipage}
&
\begin{minipage}[c]{4cm}
Couette plan 
\\
avec gradient de pression 
\\
($\partial p / \partial x \neq 0$)
\\
= ``Couette-Poiseuille''
\end{minipage}
\\

\frimage{poiseuille-cylindrique.pstex_t}
\end{minipage}
&

\[
\dpdt{\vec{u}} = \vec{0}
\]

\dpdx{\vec{u}} = \frac{\partial \vec{u}}{\partial \theta} = \vec{0}
\]
\vskip 0.5cm
\end{minipage}
&
\begin{minipage}[c]{5.5cm}
\[
\ddx{p} = C^{te}
\quad
\mbox{et} \quad
\vec{u} = u(r) \, \vec{e}_x 
\]
\[ \mbox{avec} \quad
u(r) = - \frac{1}{4 \mu} \ddx{p} \left ( R^2 - r^2\right )
\]
\vskip 0.5cm
\end{minipage}
&
\begin{minipage}[c]{4cm}
Poiseuille cylindrique
\end{minipage}
\\
\hline \hline
\begin{minipage}[c]{4.5cm}
\frimage{couette-taylor.pstex_t}
\end{minipage}
&
\begin{minipage}[c]{2cm}
\[
\dpdt{\vec{u}} = \vec{0}
\]
\[
\dpdx{\vec{u}} = \frac{\partial \vec{u}}{\partial \theta} = \vec{0}
\]
\[
\dpdx{p} = 0
\]
\vskip 0.5cm
\end{minipage}
&
\begin{minipage}[c]{5.5cm}
\[
\vec{u} = v(r) \vec{e}_\theta \quad \mbox{avec} \quad
v(r) = A r + \frac{B}{r}
\]
\[
\mbox{o\`u} \quad
A = \frac{\Omega_2 R_2^2 - \Omega_1 R_1^2}{R_2^2 - R_1^2}
\]
\[
\mbox{et} \quad
B = \frac{\left(\Omega_1 - \Omega_2\right) R_1^2 R_2^2}{R_2^2 - R_1^2}
\]
\vskip 0.5cm
\end{minipage}
&
\begin{minipage}[c]{4cm}
Couette cylindrique \\ = ``Couette-Taylor''
\end{minipage}
\\
